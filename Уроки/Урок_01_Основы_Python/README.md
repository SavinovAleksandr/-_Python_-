## Содержание урока

### 1. Переменные и типы данных

В Python переменные создаются простым присваиванием:

```python
# Числа
параметр = 10          # целое число (int)
напряжение = 220.5     # число с плавающей точкой (float)

# Строки
название_режима = "Установившийся режим"
путь_к_файлу = 'C:/RasterWin/project.rst'

# Булевы значения
расчет_завершен = True
есть_ошибка = False

# Проверка типа
print(type(параметр))        # <class 'int'>
print(type(напряжение))      # <class 'float'>
print(type(название_режима)) # <class 'str'>
```

### 2. Списки (list)

Списки - это упорядоченные коллекции элементов:

```python
# Создание списка
параметры = [10, 20, 30, 40, 50]
напряжения = [220.5, 230.0, 240.0]

# Индексация (начинается с 0)
первый_параметр = параметры[0]  # 10
последний = параметры[-1]       # 50

# Методы списков
параметры.append(60)           # добавить элемент в конец
параметры.insert(0, 5)         # вставить элемент на позицию
параметры.remove(20)           # удалить элемент
длина = len(параметры)         # длина списка

# Итерация по списку
for параметр in параметры:
    print(f"Параметр: {параметр}")
```

### 3. Словари (dict)

Словари хранят пары ключ-значение:

```python
# Создание словаря
результаты_расчета = {
    "напряжение": 220.5,
    "мощность": 1000.0,
    "режим": "установившийся",
    "устойчивость": True
}

# Доступ к значениям
напряжение = результаты_расчета["напряжение"]
режим = результаты_расчета.get("режим", "неизвестно")  # безопасный доступ

# Изменение значений
результаты_расчета["мощность"] = 1200.0
результаты_расчета["дата"] = "2024-01-15"

# Итерация по словарю
for ключ, значение in результаты_расчета.items():
    print(f"{ключ}: {значение}")

# Получение всех ключей
ключи = list(результаты_расчета.keys())
```

### 4. Условия и циклы

```python
# Условия
напряжение = 230.0

if напряжение > 240:
    print("Напряжение слишком высокое")
elif напряжение < 200:
    print("Напряжение слишком низкое")
else:
    print("Напряжение в норме")

# Цикл for
параметры = [10, 20, 30, 40, 50]
for i, параметр in enumerate(параметры):
    print(f"Индекс {i}: параметр = {параметр}")

# Цикл while
счетчик = 0
while счетчик < 5:
    print(f"Итерация {счетчик}")
    счетчик += 1

# Прерывание цикла
for параметр in параметры:
    if параметр > 30:
        break  # выход из цикла
    print(параметр)
```

### 5. Функции

```python
# Определение функции
def рассчитать_мощность(напряжение, ток):
    """
    Рассчитывает мощность по напряжению и току
    
    Аргументы:
        напряжение (float): Напряжение в вольтах
        ток (float): Ток в амперах
    
    Возвращает:
        float: Мощность в ваттах
    """
    мощность = напряжение * ток
    return мощность

# Вызов функции
мощность_1 = рассчитать_мощность(220, 5)
мощность_2 = рассчитать_мощность(напряжение=230, ток=4.5)

# Функция с параметрами по умолчанию
def запустить_расчет(режим="установившийся", точность=0.01):
    print(f"Запуск расчета: {режим}, точность: {точность}")
    return True
```

### 6. Работа с файлами

```python
# Чтение файла
with open("данные.txt", "r", encoding="utf-8") as файл:
    содержимое = файл.read()
    # или построчно:
    # строки = файл.readlines()

# Запись в файл
результаты = "Напряжение: 220.5 В\nМощность: 1000 Вт"
with open("результаты.txt", "w", encoding="utf-8") as файл:
    файл.write(результаты)

# Добавление в файл
with open("логи.txt", "a", encoding="utf-8") as файл:
    файл.write("Новая запись\n")
```
